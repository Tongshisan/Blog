## 循环递归

```js
    function flat(arr) {
        for(let i = 0, len = arr.length; i < len; i++) {
            let res = []
            if(Array.isArray(arr[i])) {
                res = [].concat(flat(arr[i]))
            } else {
                res.push(arr[i])
            }
        }

        return res
    }
```

## reduce

```js
    function flat(arr, num=1) {
        return num > 0 ? arr.reduce((acc, val) => acc.concat(Array.isArray(val) ? flat(val, num--) : val), []) : arr.slice()
    }
```
